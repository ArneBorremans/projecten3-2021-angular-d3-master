import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable, throwError } from 'rxjs';
import { map, catchError, tap, shareReplay } from 'rxjs/operators';
import { RouteLoper } from 'src/app/register-route/route-loper';
import { environment } from 'src/environments/environment';
import { AuthenticationService } from './authentication.service';
import { Loper } from 'src/app/loper/loper.model';
import { UserInfo } from 'src/app/user/profile/UserInfo';

@Injectable({
  providedIn: 'root',
})
export class LoperDataService {
  constructor(
    private http: HttpClient,
    private _authenticationService: AuthenticationService
  ) {}

  publicLopers$(page: Number): Observable<Loper[]> {
    return this.http
      .get(`${environment.apiUrl}/RouteLoper/Public/${page}`)
      .pipe(
        tap(console.log),
        map((response: any): Loper[] => response.map(Loper.fromJSON)),
        catchError((err) => this.handleError(err))
      );
  }

  loper$(): Observable<UserInfo> {
    console.log(this._authenticationService.loperId);
    return this.http
      .get(
        `${environment.apiUrl}/RouteLoper/${this._authenticationService.loperId}`
      )
      .pipe(
        tap(console.log),
        map((response: any): UserInfo => UserInfo.fromJSON(response)),
        catchError((err) => this.handleError(err))
      );
  }

  routeLoper$(): Observable<RouteLoper[]> {
    console.log(this._authenticationService.loperId);
    return this.http
      .get(
        `${environment.apiUrl}/RouteLoper/${this._authenticationService.loperId}`
      )
      .pipe(
        tap(console.log),
        map((response: any): RouteLoper[] => response.map(RouteLoper.fromJson)),
        catchError((err) => this.handleError(err))
      );
  }

  searchHiddenLopers$(
    voornaam: string,
    naam: string,
    gemeente: string
  ): Observable<Loper> {
    return this.http
      .get(
        `${environment.apiUrl}/RouteLoper/Lopers/${gemeente}/${naam}/${voornaam}`
      )
      .pipe(
        tap(console.log),
        map((response: any): Loper => Loper.fromJSON(response)),
        catchError((err) => this.handleError(err))
      );
  }

  searchPublicLopers$(
    voornaam: string,
    naam: string,
    gemeente: string
  ): Observable<Loper[]> {
    return this.http
      .get(
        `${environment.apiUrl}/RouteLoper/Public/${gemeente}/${naam}/${voornaam}`
      )
      .pipe(
        tap(console.log),
        map((response: any): Loper[] => response.map(Loper.fromJSON)),
        catchError((err) => this.handleError(err))
      );
  }

  loperID$(email: string): Observable<number> {
    return this.http
      .get(`${environment.apiUrl}/RouteLoper/LoperID/${email}`)
      .pipe(
        tap(console.log),
        catchError((err) => this.handleError(err))
      );
  }

  getCurrentRoute$(code): Observable<HuidigeLocatieResponse> {
    return this.http.get(`${environment.apiUrl}/RouteLoper/code/${code}`).pipe(
      tap(console.log),
      map(
        (response: any): HuidigeLocatieResponse =>
          HuidigeLocatieResponse.fromJson(response)
      ),
      catchError((err) => this.handleError(err))
    );
  }

  registreerVoorRoute$(
    routeID: number,
    tshirtMaat: string,
    zichtbaarheid: number
  ): Observable<RouteLoper> {
    return this.http
      .post(
        `${environment.apiUrl}/RouteLoper/New/`,
        new RouteLoper(
          routeID,
          this._authenticationService.loperId,
          tshirtMaat,
          zichtbaarheid
        ).toJson()
      )
      .pipe(
        tap(console.log),
        map((response: any): RouteLoper => RouteLoper.fromJson(response)),
        catchError((err) => this.handleError(err))
      );
  }

  ingeschreven$(routeID: number): Observable<boolean> {
    return this.http
      .get(
        `${environment.apiUrl}/RouteLoper/Ingeschreven/${routeID}/${this._authenticationService.loperId}`
      )
      .pipe(
        tap(console.log),
        catchError((err) => this.handleError(err))
      );
  }

  deleteRegistratieRoute(routeID: number) {
    return this.http
      .delete(
        `${environment.apiUrl}/RouteLoper/Delete/${routeID}/${this._authenticationService.loperId}`
      )
      .pipe(tap(console.log), catchError(this.handleError))
      .subscribe();
  }

  handleError(err: any): Observable<never> {
    let errorMessage: string;
    if (err instanceof HttpErrorResponse) {
      errorMessage = `'${err.status} ${err.statusText}' when accessing '${err.url}'`;
    } else {
      errorMessage = `an unknown error occurred ${err}`;
    }
    console.error(err);
    return throwError(errorMessage);
  }
}

class HuidigeLocatieResponse {
  constructor(
    public voornaam: string,
    public achternaam: string,
    public lat: number,
    public long: number,
    public routeId: number
  ) {}

  static fromJson(json) {
    return new HuidigeLocatieResponse(
      json.voornaam,
      json.achternaam,
      json.lat,
      json.lon,
      json.routeId
    );
  }
}
